Chat Room Routes

POST /v1/api/room/create

Description: Create a new chat room.

URL: /v1/api/room/create

Method: POST

Headers:

authorization: <token>
x-client-id: <client_id>

Request Body:

{
    "name": "string",
    "avt_url": "string",
    "user_ids": ["string (required)"]
}

Responses:

201 Created:
{
    "message": "Room created successfully",
    "metadata": {
        "name": "string",
        "avt_url": "string",
        "user_ids": ["string"],
        "is_group": boolean,
        "_id": "string",
        "createdAt": "string",
        "updatedAt": "string",
        "__v": 0
    }
}

GET /v1/api/room/messages/:room_id

Description: Get messages in a specific room.

URL: /v1/api/room/messages/:room_id

Method: GET

Headers:

authorization: <token>
x-client-id: <client_id>

Query Parameters:

page: integer
limit: integer

Responses:

200 OK:
{
    "message": "Messages retrieved successfully",
    "metadata": {
        "messages": [
            {
                "user_id": "string",
                "message": "string",
                "updated_at": "string",
                "created_at": "string",
                "status": "string",
                "room_id": "string",
                "is_group": boolean,
                "user_name": "string",
                "user_avatar": "string",
                "user_avatar_thumb": "string",
                "room_avatar": "string",
                "room_name": "string",
                "room_user_ids": ["string"]

        ],
        "currentPage": integer,
        "totalPages": integer,
        "amount": integer,
        "totalMessages": integer,
        "hasNextPage": boolean,
        "hasPrevPage": boolean
    }
}
